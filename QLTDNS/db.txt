CREATE DATABASE qlns
USE qlns
CREATE TABLE QUANLY(
    maQL VARCHAR(20) PRIMARY KEY,
    tenQL VARCHAR(50),
    diaChi TEXT,
    sdt VARCHAR(20),
    email VARCHAR(50)
)
CREATE TABLE PHONGBAN(
    maPB VARCHAR(20) PRIMARY KEY,
    tenPB VARCHAR(50),
    diaChi TEXT
)
CREATE TABLE QL_PHONGBAN(
    maQL VARCHAR(20),
    maPB VARCHAR(20),
    FOREIGN KEY(maQL) REFERENCES QUANLY(maQL),
    FOREIGN KEY (maPB) REFERENCES PHONGBAN(mapb)
) 
CREATE TABLE NHANSU(
    maNS VARCHAR(20) PRIMARY KEY,
    tenNS VARCHAR(50),
    diaChi TEXT,
    sdt VARCHAR(20),
    email VARCHAR(50),
    maPB VARCHAR(20),
    FOREIGN KEY(maPB) REFERENCES PHONGBAN(maPB)
)
CREATE TABLE NGAYCONG(
    maNC VARCHAR(20) PRIMARY KEY,
    soNC INT,
)

CREATE TABLE CHAM_CONG(
    maNS VARCHAR(20),
    maNC VARCHAR(20),
    ghiChu TEXT,
    FOREIGN KEY(maNS) REFERENCES NHANSU(maNS),
    FOREIGN KEY(maNC) REFERENCES ngaycong(maNC)
) 


CREATE TABLE QL_NHANSU(
    maQL VARCHAR(20),
    maNS VARCHAR(20) ,
    FOREIGN KEY(maQL) REFERENCES QUANLY(maQL),
    FOREIGN KEY(maNS) REFERENCES NHANSU(maNS)
)
